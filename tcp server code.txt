import socket

HOST = '0.0.0.0'
PORT = 5000

server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.bind((HOST, PORT))
server.listen(5)

print("TCP Server Started... Waiting for client")

while True:
    client_socket, addr = server.accept()
    print("Connected from", addr)

    data = client_socket.recv(1024).decode()
    print("Received:", data)

    client_socket.send("Hello from Banana Pi".encode())

    client_socket.close()
